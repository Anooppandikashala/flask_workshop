### User Helper Functions

```python
import json
import os
from werkzeug.security import generate_password_hash, check_password_hash

USERS_FILE = "data/users.json"

def load_users():
    if not os.path.exists(USERS_FILE):
        return []
    with open(USERS_FILE, "r") as f:
        return json.load(f)

def save_users(users):
    with open(USERS_FILE, "w") as f:
        json.dump(users, f, indent=2)

def find_user_by_email(email):
    return next((u for u in load_users() if u["email"] == email), None)

def find_user_by_id(user_id):
    return next((u for u in load_users() if u["id"] == user_id), None)

def register_user(name, email, password):
    users = load_users()
    if find_user_by_email(email):
        return None, "Email already registered"
    new_user = {
        "id": len(users) + 1,
        "name": name,
        "email": email,
        "password": generate_password_hash(password),
        "is_admin": False
    }
    users.append(new_user)
    save_users(users)
    return new_user, None

def verify_user(email, password):
    user = find_user_by_email(email)
    if user and check_password_hash(user["password"], password):
        return user
    return None
```

### Auth Routes in `app.py`

```python
from helpers import register_user, verify_user, find_user_by_id

# --- Auth Routes ---

@app.route("/register", methods=["GET", "POST"])
def register():
    if request.method == "POST":
        name = request.form.get("name", "").strip()
        email = request.form.get("email", "").strip().lower()
        password = request.form.get("password", "")
        confirm = request.form.get("confirm_password", "")

        if not name or not email or not password:
            flash("All fields are required.", "danger")
            return redirect(url_for("register"))

        if password != confirm:
            flash("Passwords do not match.", "danger")
            return redirect(url_for("register"))

        if len(password) < 6:
            flash("Password must be at least 6 characters.", "danger")
            return redirect(url_for("register"))

        user, error = register_user(name, email, password)
        if error:
            flash(error, "danger")
            return redirect(url_for("register"))

        session["user_id"] = user["id"]
        flash(f"Welcome, {user['name']}! Account created.", "success")
        return redirect(url_for("home"))

    return render_template("register.html")


@app.route("/login", methods=["GET", "POST"])
def login():
    if request.method == "POST":
        email = request.form.get("email", "").strip().lower()
        password = request.form.get("password", "")
        user = verify_user(email, password)
        if user:
            session["user_id"] = user["id"]
            flash(f"Welcome back, {user['name']}!", "success")
            return redirect(url_for("home"))
        else:
            flash("Invalid email or password.", "danger")

    return render_template("login.html")


@app.route("/logout")
def logout():
    session.clear()
    flash("You have been logged out.", "info")
    return redirect(url_for("home"))
```

### Protecting Routes — Login Required Decorator

```python
from functools import wraps

def login_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        if "user_id" not in session:
            flash("Please login to continue.", "warning")
            return redirect(url_for("login"))
        return f(*args, **kwargs)
    return decorated
```

```python
ORDERS_FILE = "data/orders.json"

def load_orders():
    if not os.path.exists(ORDERS_FILE):
        return []
    with open(ORDERS_FILE, "r") as f:
        return json.load(f)

def save_orders(orders):
    with open(ORDERS_FILE, "w") as f:
        json.dump(orders, f, indent=2)

def create_order(user_id, cart, address):
    from datetime import datetime
    orders = load_orders()
    items = [
        {
            "product_id": int(pid),
            "name": item["name"],
            "price": item["price"],
            "qty": item["qty"],
            "subtotal": item["price"] * item["qty"]
        }
        for pid, item in cart.items()
    ]
    total = sum(i["subtotal"] for i in items)
    order = {
        "id": len(orders) + 1,
        "user_id": user_id,
        "items": items,
        "total": total,
        "address": address,
        "status": "confirmed",
        "created_at": datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    }
    orders.append(order)
    save_orders(orders)
    return order

def get_user_orders(user_id):
    return [o for o in load_orders() if o["user_id"] == user_id]
```

### Checkout Routes

```python
from helpers import create_order, get_user_orders, find_user_by_id

@app.route("/checkout", methods=["GET", "POST"])
@login_required
def checkout():
    cart = session.get("cart", {})
    if not cart:
        flash("Your cart is empty.", "warning")
        return redirect(url_for("cart"))

    total = sum(item["price"] * item["qty"] for item in cart.values())

    if request.method == "POST":
        address = {
            "full_name":  request.form.get("full_name"),
            "address":    request.form.get("address"),
            "city":       request.form.get("city"),
            "zip_code":   request.form.get("zip_code"),
        }
        if not all(address.values()):
            flash("Please fill in all address fields.", "danger")
            return redirect(url_for("checkout"))

        user_id = session["user_id"]
        order = create_order(user_id, cart, address)
        session.pop("cart", None)   # clear cart after order
        flash(f"Order #{order['id']} placed successfully!", "success")
        return redirect(url_for("order_confirmation", order_id=order["id"]))

    return render_template("checkout.html", cart=cart, total=total)


@app.route("/order/confirmation/<int:order_id>")
@login_required
def order_confirmation(order_id):
    from helpers import load_orders
    order = next((o for o in load_orders() if o["id"] == order_id), None)
    if not order or order["user_id"] != session["user_id"]:
        return "Order not found", 404
    return render_template("order_confirmation.html", order=order)


@app.route("/orders")
@login_required
def orders():
    user_orders = get_user_orders(session["user_id"])
    user_orders.sort(key=lambda o: o["id"], reverse=True)
    return render_template("orders.html", orders=user_orders)
```

### Admin Check Decorator

```python
def admin_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        if "user_id" not in session:
            flash("Please login.", "warning")
            return redirect(url_for("login"))
        user = find_user_by_id(session["user_id"])
        if not user or not user.get("is_admin"):
            flash("Admin access required.", "danger")
            return redirect(url_for("home"))
        return f(*args, **kwargs)
    return decorated
```

```python
from flask import Flask, render_template, request, redirect, url_for, session, flash
import json, os
from datetime import datetime

app = Flask(__name__)
app.secret_key = "quickblog-secret"

POSTS_FILE = "data/posts.json"

def load_posts():
    if not os.path.exists(POSTS_FILE):
        return []
    with open(POSTS_FILE) as f:
        return json.load(f)

def save_posts(posts):
    with open(POSTS_FILE, "w") as f:
        json.dump(posts, f, indent=2)
```

#### Step 2 — Routes

```python
@app.route("/")
def index():
    posts = load_posts()
    posts.sort(key=lambda p: p["id"], reverse=True)
    return render_template("index.html", posts=posts)


@app.route("/post/<int:post_id>")
def view_post(post_id):
    post = next((p for p in load_posts() if p["id"] == post_id), None)
    if not post:
        return "Post not found", 404
    return render_template("post.html", post=post)


@app.route("/new", methods=["GET", "POST"])
def new_post():
    if "username" not in session:
        flash("Please login to write a post.", "warning")
        return redirect(url_for("login"))

    if request.method == "POST":
        title = request.form.get("title", "").strip()
        content = request.form.get("content", "").strip()

        if not title or not content:
            flash("Title and content are required.", "danger")
            return redirect(url_for("new_post"))

        posts = load_posts()
        post = {
            "id": len(posts) + 1,
            "title": title,
            "content": content,
            "author": session["username"],
            "created_at": datetime.now().strftime("%B %d, %Y")
        }
        posts.append(post)
        save_posts(posts)
        flash("Post published!", "success")
        return redirect(url_for("index"))

    return render_template("new_post.html")


@app.route("/delete/<int:post_id>", methods=["POST"])
def delete_post(post_id):
    if "username" not in session:
        return redirect(url_for("login"))
    posts = load_posts()
    post = next((p for p in posts if p["id"] == post_id), None)
    if post and post["author"] == session["username"]:
        posts = [p for p in posts if p["id"] != post_id]
        save_posts(posts)
        flash("Post deleted.", "info")
    return redirect(url_for("index"))


@app.route("/login", methods=["GET", "POST"])
def login():
    if request.method == "POST":
        username = request.form.get("username", "").strip()
        if username:
            session["username"] = username
            flash(f"Welcome, {username}!", "success")
            return redirect(url_for("index"))
        flash("Username is required.", "danger")
    return render_template("login.html")


@app.route("/logout")
def logout():
    session.clear()
    return redirect(url_for("index"))


if __name__ == "__main__":
    app.run(debug=True)
```

**4 — Word count / reading time**

Add a helper function to `app.py`:
```python
def reading_time(content):
    """Returns estimated reading time in minutes (average 200 words/min)."""
    words = len(content.split())
    minutes = max(1, round(words / 200))
    return minutes
```

Register it as a Jinja2 template filter so you can use it directly in templates:
```python
app.jinja_env.filters["reading_time"] = reading_time
```

**What to look for across all four tasks:**
- Edit: the GET loads the existing post data into the form using {% raw %}`value="{{ post.title }}"`{% endraw %}; POST saves changes back to the list

- Comments: `post["comments"]` is a list of dicts; old posts without the key use `.get("comments")` safely
- Reading time: registered as a Jinja2 filter using `app.jinja_env.filters` so it's usable with `|` in templates
